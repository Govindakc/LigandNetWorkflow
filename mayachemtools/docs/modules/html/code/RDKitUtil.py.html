  44 def GetMolName(Mol, MolNum = None):
  45     """Get molecule name.
  46     
  47     Arguments:
  48         Mol (object): RDKit molecule object.
  49         MolNum (int or None): Molecule number in input file.
  50 
  51     Returns:
  52         str : Molname corresponding to _Name property of a molecule, generated
  53             from specieid MolNum using the format "Mol%d" % MolNum, or an
  54             empty string.
  55 
  56     """
  57     
  58     MolName = ''
  59     if Mol.HasProp("_Name"):
  60         MolName = Mol.GetProp("_Name")
  61 
  62     if not len(MolName):
  63         if MolNum is not None:
  64             MolName = "Mol%d" % MolNum
  65     
  66     return MolName

  86     """

 117 def _ModifySVGForInlineEmbedding(SVGText, Base64Encoded):
 118     """Modify SVG for inline embedding into a HTML page using "img" tag
 119     along with performing base64 encoding.
 120     """
 121     
 122     # Take out all tags till the start of '<svg' tag...
 123     Pattern = re.compile("^.*<svg", re.I | re.S)
 124     SVGText = Pattern.sub("<svg", SVGText)
 125     
 126     # Add an extra space before the "width=..." tag. Otherwise, inline embedding may
 127     # cause the following XML error on some browsers due to start of the "width=..."
 128     # at the begining of the line in <svg ...> tag:
 129     #
 130     #  XML5607: Whitespace expected.
 131     #
 132     SVGText = re.sub("width='", " width='", SVGText, flags = re.I)
 133     
 134     # Take out trailing new line...
 135     SVGText = SVGText.strip()
 136 
 137     # Perform base64 encoding by turning text into byte stream using string
 138     # encode and transform byte stream returned by b64encode into a string
 139     # by string decode...
 140     #
 141     if Base64Encoded:
 142         SVGText = base64.b64encode(SVGText.encode()).decode()
 143 
 144     return SVGText

 146 def GetSVGForMolecule(Mol, Width, Height, Legend = None, AtomListToHighlight = None, BondListToHighlight = None, BoldText = True):
 147     """Get SVG image text for a molecule suitable for viewing in a browser.
 148     
 149     Arguments:
 150         Mol (object): RDKit molecule object.
 151         Width (int): Width of a molecule image in pixels.
 152         Height (int): Height of a molecule image in pixels.
 153         Legend (str): Text to display under the image.
 154         AtomListToHighlight (list): List of atoms to highlight.
 155         BondListToHighlight (list): List of bonds to highlight.
 156         BoldText (bool): Flag to make text bold in the image of molecule. 
 157 
 158     Returns:
 159         str : SVG image text for writing to a SVG file for viewing in a browser.
 160 
 161     """
 162     
 163     Mols = [Mol]
 164     
 165     MolsPerRow = 1
 166     MolWidth = Width
 167     MolHeight = Height
 168     
 169     Legends = [Legend] if Legend is not None else None
 170     AtomListsToHighlight = [AtomListToHighlight] if AtomListToHighlight is not None else None
 171     BondListsToHighLight = [BondListsToHighLight] if BondListToHighlight is not None else None
 172     
 173     return GetSVGForMolecules(Mols, MolsPerRow, MolWidth, MolHeight, Legends, AtomListsToHighlight, BondListsToHighLight, BoldText)

 175 def GetSVGForMolecules(Mols, MolsPerRow, MolWidth, MolHeight, Legends = None, AtomListsToHighlight = None, BondListsToHighlight = None, BoldText = True):
 176     """Get SVG image text for molecules suitable for viewing in a browser.
 177     
 178     Arguments:
 179         Mols (list): List of RDKit molecule objects.
 180         MolsPerRow (int): Number of molecules per row.
 181         Width (int): Width of a molecule image in pixels.
 182         Height (int): Height of a molecule image in pixels.
 183         Legends (list): List containing strings to display under images.
 184         AtomListsToHighlight (list): List of lists containing atoms to highlight
 185             for molecules.
 186         BondListsToHighlight (list): List of lists containing bonds to highlight
 187             for molecules
 188         BoldText (bool): Flag to make text bold in the image of molecules. 
 189 
 190     Returns:
 191         str : SVG image text for writing to a SVG file for viewing in a browser.
 192 
 193     """
 194     
 195     SVGText = Draw.MolsToGridImage(Mols, molsPerRow = MolsPerRow, subImgSize = (MolWidth,MolHeight), legends = Legends, highlightAtomLists = AtomListsToHighlight, highlightBondLists = BondListsToHighlight, useSVG = True)
 196     
 197     return _ModifySVGForBrowserViewing(SVGText, BoldText)
 198 
 199 def _ModifySVGForBrowserViewing(SVGText, BoldText = True):
 200     """Modify SVG for loading into a browser."""
 201     
 202     # It appears that the string 'xmlns:svg' needs to be replaced with 'xmlns' in the
 203     # SVG image string generated by RDKit. Otherwise, the image doesn't load in 
 204     # web browsers.
 205     #
 206     if re.search("xmlns:svg", SVGText, re.I):
 207         SVGText = re.sub("xmlns:svg", "xmlns", SVGText, flags = re.I)
 208     
 209     # Make text bold...
 210     if BoldText:
 211         SVGText = re.sub("font-weight:normal;", "font-weight:bold;", SVGText, flags = re.I)
 212     
 213     return SVGText

 230 def MolToBase64EncodedMolString(Mol, PropertyPickleFlags = Chem.PropertyPickleOptions.AllProps):
 231     """Encode RDkit molecule object into a base64 encoded string. The properties
 232     can be optionally excluded.
 233     
 234     The molecule is pickled using RDKit Mol.ToBinary() function before
 235     their encoding.
 236    
 237     Arguments:
 238         Mol (object): RDKit molecule object.
 239         PropertyPickleFlags: RDKit property pickle options.
 240 
 241     Returns:
 242         str : Base64 encode molecule string or None.
 243 
 244     Notes:
 245         The following property pickle flags are currently available in RDKit:
 246             
 247             Chem.PropertyPickleOptions.NoProps
 248             Chem.PropertyPickleOptions.MolProps
 249             Chem.PropertyPickleOptions.AtomProps
 250             Chem.PropertyPickleOptions.BondProps
 251             Chem.PropertyPickleOptions.PrivateProps
 252             Chem.PropertyPickleOptions.AllProps
 253 
 254     """
 255 
 256     return None if Mol is None else base64.b64encode(Mol.ToBinary(PropertyPickleFlags)).decode()

 258 def MolFromBase64EncodedMolString(EncodedMol):

 261     Arguments:
 262         str: Base64 encoded molecule string.
 263 
 264     Returns:

 271 def GenerateBase64EncodedMolStrings(Mols, PropertyPickleFlags = Chem.PropertyPickleOptions.AllProps):
 272     """Setup an iterator for generating base64 encoded molecule string
 273     from a RDKit molecule iterator. The iterator returns a list containing
 274     a molecule index and encoded molecule string or None.
 275     
 276     The molecules are pickled using RDKit Mol.ToBinary() function
 277     before their encoding.
 278     
 279     Arguments:
 280         iterator: RDKit molecules iterator.
 281         PropertyFlags: RDKit property pickle options.
 282 
 283     Returns:
 284         object : Base64 endcoded molecules iterator. The iterator returns a
 285             list containing a molecule index and an encoded molecule string
 286             or None.
 287 
 288     Notes:
 289         The following property pickle flags are currently available in RDKit:
 290             
 291             Chem.PropertyPickleOptions.NoProps
 292             Chem.PropertyPickleOptions.MolProps
 293             Chem.PropertyPickleOptions.AtomProps
 294             Chem.PropertyPickleOptions.BondProps
 295             Chem.PropertyPickleOptions.PrivateProps
 296             Chem.PropertyPickleOptions.AllProps
 297 
 298     Examples:
 299 
 300         EncodedMolsInfo = GenerateBase64EncodedMolStrings(Mols)
 301         for MolIndex, EncodedMol in EncodedMolsInfo:
 302             if EncodeMol is not None:
 303                 Mol = MolFromBase64EncodedMolString(EncodedMol)
 304 
 305     """
 306     for MolIndex, Mol in enumerate(Mols):
 307         yield [MolIndex, None] if Mol is None else [MolIndex, MolToBase64EncodedMolString(Mol, PropertyPickleFlags)]

 309 def ReadAndValidateMolecules(FileName, **KeyWordArgs):
 310     """Read molecules from an input file, validate all molecule objects, and return
 311     a list of valid and non-valid molecule objects along with their counts.
 312     
 313     Arguments:
 314         FileName (str): Name of a file with complete path.
 315         **KeyWordArgs (dictionary) : Parameter name and value pairs for reading and
 316             processing molecules.
 317 
 318     Returns:
 319         list : List of valid RDKit molecule objects.
 320         int : Number of total molecules in input file. 
 321         int : Number of valid molecules in input file. 
 322 
 323     Notes:
 324         The file extension is used to determine type of the file and set up an appropriate
 325         file reader.
 326 
 327     """
 328 
 329     AllowEmptyMols = True
 330     if "AllowEmptyMols" in KeyWordArgs:
 331         AllowEmptyMols = KeyWordArgs["AllowEmptyMols"]
 332     
 333     Mols = ReadMolecules(FileName, **KeyWordArgs)
 334 
 335     if AllowEmptyMols:
 336         ValidMols = [Mol for Mol in Mols if Mol is not None]
 337     else:
 338         ValidMols = []
 339         MolCount = 0
 340         for Mol in Mols:
 341             MolCount += 1
 342             if Mol is None:
 343                 continue
 344             
 345             if IsMolEmpty(Mol):
 346                 MolName = GetMolName(Mol, MolCount)
 347                 MiscUtil.PrintWarning("Ignoring empty molecule: %s" % MolName)
 348                 continue
 349             
 350             ValidMols.append(Mol)
 351             
 352     MolCount = len(Mols)
 353     ValidMolCount = len(ValidMols)
 354 
 355     return (ValidMols, MolCount, ValidMolCount)

 357 def ReadMolecules(FileName, **KeyWordArgs):
 358     """Read molecules from an input file without performing any validation
 359     and creation of molecule objects.
 360     
 361     Arguments:
 362         FileName (str): Name of a file with complete path.
 363         **KeyWordArgs (dictionary) : Parameter name and value pairs for reading and
 364             processing molecules.
 365 
 366     Returns:
 367         list : List of RDKit molecule objects.
 368 
 369     Notes:
 370         The file extension is used to determine type of the file and set up an appropriate
 371         file reader.
 372 
 373     """
 374 
 375     # Set default values for possible arguments...
 376     ReaderArgs = {"Sanitize": True, "RemoveHydrogens": True, "StrictParsing": True,  "SMILESDelimiter" : ' ', "SMILESColumn": 1, "SMILESNameColumn": 2, "SMILESTitleLine": True }
 377 
 378     # Set specified values for possible arguments...
 379     for Arg in ReaderArgs:
 380         if Arg in KeyWordArgs:
 381             ReaderArgs[Arg] = KeyWordArgs[Arg]
 382 
 383     # Modify specific valeus for SMILES...
 384     if MiscUtil.CheckFileExt(FileName, "smi csv tsv txt"):
 385         Args = ["Sanitize", "SMILESTitleLine"]
 386         for Arg in Args:
 387             if ReaderArgs[Arg] is True:
 388                 ReaderArgs[Arg] = 1
 389             else:
 390                 ReaderArgs[Arg] = 0
 391     
 392     Mols = []
 393     if MiscUtil.CheckFileExt(FileName, "sdf sd"):
 394         return ReadMoleculesFromSDFile(FileName, ReaderArgs["Sanitize"], ReaderArgs["RemoveHydrogens"], ReaderArgs['StrictParsing'])
 395     elif MiscUtil.CheckFileExt(FileName, "mol"):
 396         return ReadMoleculesFromMolFile(FileName, ReaderArgs["Sanitize"], ReaderArgs["RemoveHydrogens"], ReaderArgs['StrictParsing'])
 397     elif MiscUtil.CheckFileExt(FileName, "mol2"):
 398         return ReadMoleculesFromMol2File(FileName, ReaderArgs["Sanitize"], ReaderArgs["RemoveHydrogens"])
 399     elif MiscUtil.CheckFileExt(FileName, "pdb"):
 400         return ReadMoleculesFromPDBFile(FileName, ReaderArgs["Sanitize"], ReaderArgs["RemoveHydrogens"])
 401     elif MiscUtil.CheckFileExt(FileName, "smi txt csv tsv"):
 402         SMILESColumnIndex = ReaderArgs["SMILESColumn"] - 1
 403         SMILESNameColumnIndex = ReaderArgs["SMILESNameColumn"] - 1
 404         return ReadMoleculesFromSMILESFile(FileName, ReaderArgs["SMILESDelimiter"], SMILESColumnIndex, SMILESNameColumnIndex, ReaderArgs["SMILESTitleLine"], ReaderArgs["Sanitize"])
 405     else:
 406         MiscUtil.PrintWarning("RDKitUtil.ReadMolecules: Non supported file type: %s" % FileName)
 407     
 408     return Mols

 494 def MoleculesWriter(FileName, **KeyWordArgs):
 495     """Set up a molecule writer.
 496     
 497     Arguments:
 498         FileName (str): Name of a file with complete path.
 499         **KeyWordArgs (dictionary) : Parameter name and value pairs for writing and
 500             processing molecules.
 501 
 502     Returns:
 503         RDKit object : Molecule writer.
 504 
 505     Notes:
 506         The file extension is used to determine type of the file and set up an appropriate
 507         file writer.
 508 
 509     """
 510     
 511     # Set default values for possible arguments...
 512     WriterArgs = {"Compute2DCoords" : False, "Kekulize": False, "SMILESDelimiter" : ' ', "SMILESIsomeric": True, "SMILESTitleLine": True }
 513 
 514     # Set specified values for possible arguments...
 515     for Arg in WriterArgs:
 516         if Arg in KeyWordArgs:
 517             WriterArgs[Arg] = KeyWordArgs[Arg]
 518     
 519     Writer = None
 520     if MiscUtil.CheckFileExt(FileName, "sdf sd"):
 521         Writer = Chem.SDWriter(FileName)
 522         if WriterArgs["Kekulize"]:
 523             Writer.SetKekulize(True)
 524     elif MiscUtil.CheckFileExt(FileName, "pdb"):
 525         Writer = Chem.PDBWriter(FileName)
 526     elif MiscUtil.CheckFileExt(FileName, "smi"):
 527         # Text for the name column in the title line. Blank indicates not to include name column
 528         # in the output file...
 529         NameHeader = 'Name'
 530         Writer = Chem.SmilesWriter(FileName, delimiter = WriterArgs["SMILESDelimiter"], nameHeader = NameHeader, includeHeader = WriterArgs["SMILESTitleLine"],  isomericSmiles = WriterArgs["SMILESIsomeric"], kekuleSmiles = WriterArgs["Kekulize"])
 531     else:
 532         MiscUtil.PrintWarning("RDKitUtil.WriteMolecules: Non supported file type: %s" % FileName)
 533     
 534     return Writer

 536 def WriteMolecules(FileName, Mols, **KeyWordArgs):
 537     """Write molecules to an output file.
 538     
 539     Arguments:
 540         FileName (str): Name of a file with complete path.
 541         Mols (list): List of RDKit molecule objects. 
 542         **KeyWordArgs (dictionary) : Parameter name and value pairs for writing and
 543             processing molecules.
 544 
 545     Returns:
 546         int : Number of total molecules.
 547         int : Number of processed molecules written to output file.
 548 
 549     Notes:
 550         The file extension is used to determine type of the file and set up an appropriate
 551         file writer.
 552 
 553     """
 554     
 555     Compute2DCoords = False
 556     if "Compute2DCoords" in KeyWordArgs:
 557         Compute2DCoords = KeyWordArgs["Compute2DCoords"]
 558         
 559     MolCount = len(Mols)
 560     ProcessedMolCount = 0
 561     
 562     Writer = MoleculesWriter(FileName, **KeyWordArgs)
 563     
 564     if Writer is None:
 565         return (MolCount, ProcessedMolCount)
 566     
 567     for Mol in Mols:
 568         if Mol is None:
 569             continue
 570         
 571         ProcessedMolCount += 1
 572         if Compute2DCoords:
 573             AllChem.Compute2DCoords(Mol)
 574         
 575         Writer.write(Mol)
 576     
 577     Writer.close()
 578     
 579     return (MolCount, ProcessedMolCount)
